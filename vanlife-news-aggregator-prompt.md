```
‚ùå https://news.google.com/rss/articles/CBMivAFBVV95cUxQQ2J4MGNn...
```

### –†–µ—à–µ–Ω–∏–µ
–°–æ–∑–¥–∞—Ç—å –∫–ª–∞—Å—Å `GoogleNewsUrlDecoder` —Å –¥–≤—É–º—è –º–µ—Ç–æ–¥–∞–º–∏ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏—è:

### –†–µ–∞–ª–∏–∑–∞—Ü–∏—è PHP

```php
<?php
// –§–∞–π–ª: src/Service/GoogleNewsUrlDecoder.php

namespace App\Service;

/**
 * –î–µ–∫–æ–¥–µ—Ä URL –∏–∑ Google News RSS
 * 
 * Google News –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —Å—Å—ã–ª–∫–∏ –≤–∏–¥–∞:
 * https://news.google.com/rss/articles/CBMi...
 * 
 * –≠—Ç–æ—Ç –∫–ª–∞—Å—Å –∏–∑–≤–ª–µ–∫–∞–µ—Ç –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—ã–π URL —Å—Ç–∞—Ç—å–∏.
 * 
 * –û—Å–Ω–æ–≤–∞–Ω –Ω–∞:
 * - https://github.com/SSujitX/google-news-url-decoder
 * - https://gist.github.com/huksley/bc3cb046157a99cd9d1517b32f91a99e
 */
class GoogleNewsUrlDecoder
{
    private const GOOGLE_NEWS_URL_PREFIX = 'https://news.google.com/';
    private const ARTICLES_PATH = '/rss/articles/';
    private const READ_PATH = '/read/';
    
    private LoggerService $logger;
    private int $requestDelay = 1000; // ms –º–µ–∂–¥—É –∑–∞–ø—Ä–æ—Å–∞–º–∏ –∫ Google
    private ?string $lastRequestTime = null;
    
    public function __construct(LoggerService $logger)
    {
        $this->logger = $logger;
    }
    
    /**
     * –î–µ–∫–æ–¥–∏—Ä—É–µ—Ç Google News URL –≤ –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—ã–π URL
     * 
     * @param string $googleNewsUrl URL –∏–∑ Google News RSS
     * @return string|null –û—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—ã–π URL –∏–ª–∏ null –ø—Ä–∏ –æ—à–∏–±–∫–µ
     */
    public function decode(string $googleNewsUrl): ?string
    {
        // –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ —ç—Ç–æ Google News URL
        if (!$this->isGoogleNewsUrl($googleNewsUrl)) {
            return $googleNewsUrl; // –£–∂–µ –æ–±—ã—á–Ω—ã–π URL
        }
        
        // –ò–∑–≤–ª–µ–∫–∞–µ–º –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω—É—é —á–∞—Å—Ç—å
        $encodedPart = $this->extractEncodedPart($googleNewsUrl);
        if (!$encodedPart) {
            $this->logger->warning('GoogleNewsDecoder', 'Cannot extract encoded part', [
                'url' => $googleNewsUrl
            ]);
            return null;
        }
        
        // –ú–µ—Ç–æ–¥ 1: –ü—Ä–æ–±—É–µ–º –±—ã—Å—Ç—Ä–æ–µ base64 –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–µ
        $decoded = $this->decodeBase64($encodedPart);
        if ($decoded && filter_var($decoded, FILTER_VALIDATE_URL)) {
            return $decoded;
        }
        
        // –ú–µ—Ç–æ–¥ 2: –ó–∞–ø—Ä–æ—Å –∫ Google –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏—è
        $decoded = $this->decodeViaGoogleApi($encodedPart);
        if ($decoded) {
            return $decoded;
        }
        
        // –ú–µ—Ç–æ–¥ 3: Fallback ‚Äî –¥–µ–ª–∞–µ–º HTTP –∑–∞–ø—Ä–æ—Å –∏ —Å–ª–µ–¥—É–µ–º —Ä–µ–¥–∏—Ä–µ–∫—Ç–∞–º
        $decoded = $this->decodeViaRedirect($googleNewsUrl);
        if ($decoded) {
            return $decoded;
        }
        
        $this->logger->error('GoogleNewsDecoder', 'All decode methods failed', [
            'url' => $googleNewsUrl
        ]);
        
        return null;
    }
    
    /**
     * –ü–∞–∫–µ—Ç–Ω–æ–µ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–µ URL (–æ–ø—Ç–∏–º–∏–∑–∏—Ä–æ–≤–∞–Ω–æ –¥–ª—è –º–Ω–æ–∂–µ—Å—Ç–≤–∞ —Å—Å—ã–ª–æ–∫)
     * 
     * @param array $urls –ú–∞—Å—Å–∏–≤ Google News URL
     * @return array –ê—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–π –º–∞—Å—Å–∏–≤ [original_url => decoded_url]
     */
    public function decodeBatch(array $urls): array
    {
        $results = [];
        
        foreach ($urls as $url) {
            $results[$url] = $this->decode($url);
            
            // –ó–∞–¥–µ—Ä–∂–∫–∞ –º–µ–∂–¥—É –∑–∞–ø—Ä–æ—Å–∞–º–∏ –¥–ª—è –∏–∑–±–µ–∂–∞–Ω–∏—è rate limiting
            if ($this->requestDelay > 0) {
                usleep($this->requestDelay * 1000);
            }
        }
        
        return $results;
    }
    
    /**
     * –ü—Ä–æ–≤–µ—Ä—è–µ—Ç, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ URL —Å—Å—ã–ª–∫–æ–π Google News
     */
    public function isGoogleNewsUrl(string $url): bool
    {
        return str_starts_with($url, self::GOOGLE_NEWS_URL_PREFIX);
    }
    
    /**
     * –ò–∑–≤–ª–µ–∫–∞–µ—Ç –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω—É—é —á–∞—Å—Ç—å –∏–∑ URL
     */
    private function extractEncodedPart(string $url): ?string
    {
        $parsed = parse_url($url);
        if (!$parsed || !isset($parsed['path'])) {
            return null;
        }
        
        $path = $parsed['path'];
        
        // –§–æ—Ä–º–∞—Ç: /rss/articles/CBMi...
        if (str_contains($path, '/articles/')) {
            $parts = explode('/articles/', $path);
            if (isset($parts[1])) {
                // –£–±–∏—Ä–∞–µ–º query –ø–∞—Ä–∞–º–µ—Ç—Ä—ã
                return explode('?', $parts[1])[0];
            }
        }
        
        // –§–æ—Ä–º–∞—Ç: /read/CBMi...
        if (str_contains($path, '/read/')) {
            $parts = explode('/read/', $path);
            if (isset($parts[1])) {
                return explode('?', $parts[1])[0];
            }
        }
        
        return null;
    }
    
    /**
     * –ú–µ—Ç–æ–¥ 1: –î–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–µ —á–µ—Ä–µ–∑ base64
     * 
     * –†–∞–±–æ—Ç–∞–µ—Ç –¥–ª—è URL —Ñ–æ—Ä–º–∞—Ç–∞ CBMi... (—Å—Ç–∞—Ä—ã–π —Ñ–æ—Ä–º–∞—Ç Google News)
     * –ù–æ–≤—ã–π —Ñ–æ—Ä–º–∞—Ç (2024+) —Ç—Ä–µ–±—É–µ—Ç API-–∑–∞–ø—Ä–æ—Å
     */
    private function decodeBase64(string $encoded): ?string
    {
        try {
            // –î–æ–±–∞–≤–ª—è–µ–º padding –¥–ª—è base64
            $padded = $encoded . str_repeat('=', (4 - strlen($encoded) % 4) % 4);
            
            // –î–µ–∫–æ–¥–∏—Ä—É–µ–º base64 URL-safe
            $decoded = base64_decode(strtr($padded, '-_', '+/'), true);
            if ($decoded === false) {
                return null;
            }
            
            // –ò—â–µ–º URL –≤ –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
            // –§–æ—Ä–º–∞—Ç: \x08\x13"<length><url>\xd2\x01\x00
            
            // –ü–∞—Ç—Ç–µ—Ä–Ω –¥–ª—è –∏–∑–≤–ª–µ—á–µ–Ω–∏—è URL
            if (preg_match('/https?:\/\/[^\x00-\x1f\x7f-\xff]+/', $decoded, $matches)) {
                $url = $matches[0];
                
                // –û—á–∏—â–∞–µ–º –æ—Ç –º—É—Å–æ—Ä–∞ –≤ –∫–æ–Ω—Ü–µ
                $url = preg_replace('/[\x00-\x1f\x7f-\xff].*$/', '', $url);
                
                // –ü—Ä–æ–≤–µ—Ä—è–µ–º –≤–∞–ª–∏–¥–Ω–æ—Å—Ç—å
                if (filter_var($url, FILTER_VALIDATE_URL)) {
                    return $url;
                }
            }
            
            // –ê–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω—ã–π –º–µ—Ç–æ–¥: –∏—â–µ–º –ø–æ —Å—Ç—Ä—É–∫—Ç—É—Ä–µ protobuf
            $bytes = array_values(unpack('C*', $decoded));
            
            // –ü—Ä–æ–ø—É—Å–∫–∞–µ–º header bytes (–æ–±—ã—á–Ω–æ \x08\x13")
            $startIndex = 0;
            for ($i = 0; $i < min(10, count($bytes)); $i++) {
                if ($bytes[$i] === 0x22) { // "
                    $startIndex = $i + 1;
                    break;
                }
            }
            
            if ($startIndex > 0 && $startIndex < count($bytes)) {
                $length = $bytes[$startIndex];
                
                // –û–±—Ä–∞–±–æ—Ç–∫–∞ varint –¥–ª—è –¥–ª–∏–Ω—ã > 127
                if ($length >= 0x80 && $startIndex + 1 < count($bytes)) {
                    $length = ($length & 0x7f) | ($bytes[$startIndex + 1] << 7);
                    $startIndex++;
                }
                
                $startIndex++;
                $endIndex = min($startIndex + $length, count($bytes));
                
                $urlBytes = array_slice($bytes, $startIndex, $endIndex - $startIndex);
                $url = implode('', array_map('chr', $urlBytes));
                
                // –û–±—Ä–µ–∑–∞–µ–º –ø–æ –ø–µ—Ä–≤–æ–º—É –Ω–µ–≤–∞–ª–∏–¥–Ω–æ–º—É —Å–∏–º–≤–æ–ª—É
                if (preg_match('/^(https?:\/\/[^\x00-\x1f\x7f-\xff]+)/', $url, $matches)) {
                    if (filter_var($matches[1], FILTER_VALIDATE_URL)) {
                        return $matches[1];
                    }
                }
            }
            
            return null;
            
        } catch (\Throwable $e) {
            $this->logger->debug('GoogleNewsDecoder', 'Base64 decode failed', [
                'error' => $e->getMessage()
            ]);
            return null;
        }
    }
    
    /**
```
